<script> 
const listaDeValores = [5, 10, 15, 20, 25];

function somarValores(lista) {
  let soma = 0;

  for (let i = 0; i < lista.length; i++) {
    soma += lista[i];
  }

  return soma;
}

const resultado = somarValores(listaDeValores);
document.write(`A soma dos valores da lista é: ${resultado}`);



</script>
